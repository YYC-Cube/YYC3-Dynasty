## 1.1 核心实体定义

### Edict（敕令）

```typescript
interface Edict {
  id: string;
  title: string;
  content: string;
  
  // 类型与优先级
  type: 'edict_sys' | 'edict_norm' | 'edict_fast' | 'edict_dept';
  priority: 'P0' | 'P1' | 'P2' | 'P3';
  
  // 流转状态
  stage: EdictStage;
  stageHistory: StageTransition[];
  
  // 印章记录
  seals: SealRecord[];
  
  // 关联
  creatorId: string;
  assigneeIds: string[];
  
  // 时间戳
  createdAt: Date;
  updatedAt: Date;
  deadline?: Date;
  
  // 状态
  status: 'active' | 'completed' | 'cancelled' | 'revoked';
}

type EdictStage = 
  | 'draft'       // 初稿 - 白麻
  | 'sorting'     // 分拣 - 明堂
  | 'proposed'    // 草拟 - 黄纸
  | 'reviewed'    // 审议 - 绯绫
  | 'dispatched'  // 派发 - 尚书
  | 'executing'   // 执行 - 六部
  | 'reporting';  // 回奏 - 定鼎门
```

### Skill（技能）

```typescript
interface Skill {
  id: string;
  name: string;
  description: string;
  
  // 层级
  layer: 'L1' | 'L2';
  
  // L1专属
  dynasty?: string;           // 关联朝代
  culturalSource?: string;    // 典故出处
  
  // L2专属
  department?: string;        // 归属部门
  
  // 类型
  type: 'active' | 'passive';
  category: SkillCategory;
  
  // 效果
  baseEffect: number;         // 基础加成 (如 1.10 = +10%)
  buffCoefficient?: number;   // L1文化加成系数
  
  // 资源消耗（主动技能）
  cost?: {
    decreePoints: number;      // 政令值消耗
    cooldownSeconds: number;   // 冷却时间(秒)
  };
  
  // 状态
  status: SkillStatus;
  unlockedAt?: Date;
  lastTriggeredAt?: Date;
}

type SkillStatus = 
  | 'locked'
  | 'unlocked'
  | 'active'
  | 'cooling_down'
  | 'error';

type SkillCategory = 
  | 'governance'    // 治理
  | 'ritual'        // 礼乐
  | 'order'         // 秩序
  | 'achievement'   // 功业
  | 'strategy'      // 文韬
  | 'culture'       // 文事
  | 'innovation'    // 革新
  | 'construction'  // 营建
  | 'prosperity'    // 盛世
  | 'transition'    // 更迭
  | 'warning';      // 警世
```

### Agent（职司角色）

```typescript
interface Agent {
  id: string;
  code: string;              // 代号: Emperor, Taizi, Dragon...
  name: string;              // 角色名
  title: string;             // 职衔
  
  // 组织归属
  department: Department;
  layer: 'decision' | 'relay' | 'review' | 'execute' | 'assist';
  
  // 能力
  skills: string[];          // 拥有的技能ID列表
  
  // 状态
  status: 'online' | 'busy' | 'offline';
  currentTaskId?: string;
  
  // 统计
  stats: {
    tasksCompleted: number;
    averageResponseTime: number;  // ms
    qualityScore: number;          // 0-100
  };
}
```

## 1.2 状态管理接口

```typescript
// 全局状态结构
interface DynastyState {
  // 当前会话
  currentSession: {
    activeDynasty: string | null;    // 当前激活的朝代
    activeSkills: string[];           // 已激活的技能ID列表
    userRole: UserRole;               // 用户角色
  };
  
  // 敕令管理
  edicts: {
    active: Edict[];                  // 进行中的敕令
    completed: Edict[];               // 已完成的敕令
    drafts: Edict[];                  // 草稿
  };
  
  // 资源
  resources: {
    decreePoints: number;             // 政令值
    tokenQuota: Record<string, number>; // 各部门令牌余量
  };
  
  // 系统
  system: {
    healthStatus: 'healthy' | 'warning' | 'critical';
    lastSyncTime: Date;
    version: string;
  };
}
```
